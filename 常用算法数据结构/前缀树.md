class Trie {
    String word;
    boolean isWord;
    Map<Character, Trie> child;

    public Trie() {
        this.word = "";
        this.isWord = false;
        this.child = new HashMap<>();
    }

    public static void add(Trie root, String s) {
        Trie node = root;
        for (char c : s.toCharArray()) {
            if (!node.child.containsKey(c))
                node.child.put(c, new Trie());
            node = node.child.get(c);
        }
        node.isWord = true;
        node.word = s;
    }

    public static boolean match(Trie root, String s) {
        Trie node = root;
        for (char c : s.toCharArray()) {
            if (!node.child.containsKey(c))
                return false;
            node = node.child.get(c);
        }
        return node.isWord;
    }

    public static boolean prefix(Trie root, String pre) {
        Trie node = root;
        for (char c : pre.toCharArray()) {
            if (!node.child.containsKey(c))
                return false;
            node = node.child.get(c);
        }
        return true;
    }
}
